🛠️ ESP32 Arduino Code

// LifeCode AI — ESP32 Sensor Sender
// Upload this to your ESP32 board

#define PULSE_PIN  36    // Analog GPIO for Pulse Sensor
#define GSR_PIN    39    // Analog GPIO for GSR Sensor

void setup() {
  Serial.begin(115200);
}

void loop() {
  int   pulseRaw = analogRead(PULSE_PIN);
  int   gsrRaw   = analogRead(GSR_PIN);

  // Convert raw values
  int   heartRate  = map(pulseRaw, 0, 4095, 50, 120);
  float gsr        = gsrRaw / 100.0;
  float temperature = 36.5 + random(-5, 10) / 100.0;

  // Send CSV: heart_rate,gsr,temperature
  Serial.print(heartRate);
  Serial.print(",");
  Serial.print(gsr, 1);
  Serial.print(",");
  Serial.println(temperature, 1);

  delay(1000);  // Send every 1 second
}
